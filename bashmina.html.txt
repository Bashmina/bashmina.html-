                } else {

        function viewProduct(productId) {
            const products = getProducts();
            const product = products[productId];
            if (!product) return;

            currentProductId = productId;
            
            // Set product details
            document.getElementById('productDetailTitle').textContent = product.name;
            
            if (product.oldPrice) {
                document.getElementById('productDetailPrice').innerHTML = `
                    <span class="old-price-large">${product.oldPrice.toLocaleString()} ر.ي</span>
                    <span>${product.price.toLocaleString()} ر.ي</span>
                    <span style="background-color: var(--danger); color: white; padding: 3px 10px; border-radius: 20px; font-size: 0.9rem; margin-right: 10px;">خصم ${product.discount}%</span>
                `;
            } else {
                document.getElementById('productDetailPrice').innerHTML = `
                    <span>${product.price.toLocaleString()} ر.ي</span>
                `;
            }
            
            document.getElementById('productDetailCategory').textContent = product.category;
            document.getElementById('productDetailQuantity').textContent = product.quantity;
            document.getElementById('productDetailDescription').textContent = product.description;
            document.getElementById('productQuantityInput').value = 1;
            
            // Set product images
            const mainImage = document.getElementById('productMainImage');
            const thumbnailsContainer = document.querySelector('.thumbnail-container');
            
            mainImage.src = product.images[0];
            thumbnailsContainer.innerHTML = '';
            
            product.images.forEach((image, index) => {
                const thumbnail = document.createElement('img');
                thumbnail.src = image;
                thumbnail.className = 'thumbnail' + (index === 0 ? ' active' : '');
                thumbnail.onclick = function() { changeMainImage(this); };
                thumbnailsContainer.appendChild(thumbnail);
            });
            
            // Set color options
            const colorOptions = document.querySelector('.color-options');
            colorOptions.innerHTML = '';
            
            product.colors.forEach((color, index) => {
                const colorOption = document.createElement('div');
                colorOption.className = 'color-option' + (index === 0 ? ' selected' : '');
                colorOption.title = color;
                
                // Map color names to actual colors
                let colorValue;
                switch(color.toLowerCase()) {
                    case 'أبيض': colorValue = '#ffffff'; break;
                    case 'أسود': colorValue = '#000000'; break;
                    case 'أزرق': colorValue = '#1e40af'; break;
                    case 'بني': colorValue = '#9a3412'; break;
                    case 'رمادي': colorValue = '#6b7280'; break;
                    case 'بيج': colorValue = '#d4a373'; break;
                    default: colorValue = '#ffffff';
                }
                
                colorOption.style.backgroundColor = colorValue;
                if (color.toLowerCase() === 'أبيض') {
                    colorOption.style.border = '1px solid #ddd';
                }
                
                colorOption.onclick = function() {
                    document.querySelectorAll('.color-option').forEach(opt => opt.classList.remove('selected'));
                    this.classList.add('selected');
                };
                
                colorOptions.appendChild(colorOption);
            });
            
            openModal('productDetailsModal');
        }

        function sendVerificationCode() {
            const name = document.getElementById('fullName').value;
            const phone = document.getElementById('phoneNumber').value;
            const country = document.getElementById('country').value;
            const region = document.getElementById('region').value;
            
            if (!name || !phone || !country || !region) {
                showToast('يرجى ملء جميع الحقول المطلوبة', 'error');
                return;
            }
            
            // Generate random 5-digit code
            verificationCode = Math.floor(10000 + Math.random() * 90000).toString();
            
            // In a real app, you would send this code via WhatsApp
            console.log(`Verification code for ${phone}: ${verificationCode}`);
            
            // Show verification form
            newUserForm.style.display = 'none';
            verificationForm.style.display = 'block';
            
            // Auto-focus first input
            document.querySelector('.verification-code input').focus();
            
            // Set up auto-tab between inputs
            const inputs = document.querySelectorAll('.verification-code input');
            inputs.forEach((input, index) => {
                input.addEventListener('input', () => {
                    if (input.value.length === 1 && index < inputs.length - 1) {
                        inputs[index + 1].focus();
                    }
                });
                
                input.addEventListener('keydown', (e) => {
                    if (e.key === 'Backspace' && input.value.length === 0 && index > 0) {
                        inputs[index - 1].focus();
                    }
                });
            });
        }

        function verifyCode() {
            const enteredCode = Array.from(document.querySelectorAll('.verification-code input'))
                .map(input => input.value)
                .join('');
            
            if (enteredCode === verificationCode) {
                // Code is correct, register user
                const name = document.getElementById('fullName').value;
                const phone = document.getElementById('phoneNumber').value;
                const country = document.getElementById('country').value;
                const region = document.getElementById('region').value;
                
                // Save user
                const user = { name, phone, country, region };
                localStorage.setItem('user', JSON.stringify(user));
                localStorage.setItem('lastLogin', new Date().getTime());
                
                // Set as current user
                currentUser = user;
                
                // Show success
                showToast('تم تسجيل الدخول بنجاح');
                closeModal('userLoginModal');
            } else {
                showToast('رمز التحقق غير صحيح', 'error');
            }
        }
